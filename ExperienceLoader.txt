local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
   Name = "Experience GUI",
   LoadingTitle = "Loading Experience GUI",
   LoadingSubtitle = "by ARandomDumDev",
   ConfigurationSaving = {
      Enabled = true,
      FolderName = nil,
      FileName = "ExperienceGuiConfig"
   },
   Discord = {
      Enabled = false,
      Invite = "noinvitelink",
      RemoteEventAccess = true
   },
   KeySystem = false,
   KeySettings = {
      Title = "Untitled",
      Subtitle = "Subtitle",
      Note = "Note",
      FileName = "Key",
      SaveKey = true,
      SavePath = "roblox/settings",
      ModifiedCallback = function(Key) end
   }
})

local Tab = Window:CreateTab("Main", 4483362458)

local Section = Tab:CreateSection("Controls")

Tab:CreateButton({
   Name = "Give Experience",
   Callback = function()
      local player = game.Players.LocalPlayer
      local leaderstats = player:WaitForChild("leaderstats")
      local experience = leaderstats:WaitForChild("experience")
      experience.Value = experience.Value + 100
   end,
})

Tab:CreateButton({
   Name = "Reset Experience",
   Callback = function()
      local player = game.Players.LocalPlayer
      local leaderstats = player:WaitForChild("leaderstats")
      local experience = leaderstats:WaitForChild("experience")
      experience.Value = 0
   end,
})

Tab:CreateButton({
   Name = "Set Experience to 1000",
   Callback = function()
      local player = game.Players.LocalPlayer
      local leaderstats = player:WaitForChild("leaderstats")
      local experience = leaderstats:WaitForChild("experience")
      experience.Value = 1000
   end,
})

Tab:CreateButton({
   Name = "Multiply Experience by 2",
   Callback = function()
      local player = game.Players.LocalPlayer
      local leaderstats = player:WaitForChild("leaderstats")
      local experience = leaderstats:WaitForChild("experience")
      experience.Value = experience.Value * 2
   end,
})

Rayfield:LoadConfiguration()
