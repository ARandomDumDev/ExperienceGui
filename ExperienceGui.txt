-- Load Rayfield library
local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Create main window with key system (password gating)
local Window = Rayfield:CreateWindow({
    Name = "ExperienceGui",
    Icon = 0,
    LoadingTitle = "Loading...",
    LoadingSubtitle = "Please wait",
    ShowText = "ExperienceGui",
    ToggleUIKeybind = "K",
    Theme = "Default",
    KeySystem = true,
    KeySettings = {
        Title = "Password Required",
        Subtitle = "Enter password to continue",
        FileName = "CoolDownKey",
        SaveKey = false,
        GrabKeyFromSite = false,
        Key = {"CoolDownExperience"}
    }
})

-- Get player and character info
local player = game:GetService("Players").LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")

-- Home Tab (welcome message and credits)
local homeTab = Window:CreateTab("Home", 0)
homeTab:CreateParagraph({
    Title = "Welcome",
    Content = "Welcome to the Experience GUI!\nScript by CoolDown :D"
})

-- Client Info Tab (player name, display name, user ID)
local clientTab = Window:CreateTab("Client Info", 0)
clientTab:CreateLabel("Name: " .. player.Name, 0, Color3.fromRGB(255,255,255), false)
clientTab:CreateLabel("Display Name: " .. player.DisplayName, 0, Color3.fromRGB(255,255,255), false)
clientTab:CreateLabel("User ID: " .. player.UserId, 0, Color3.fromRGB(255,255,255), false)

-- Executor Info Tab (identifyexecutor result)
local execTab = Window:CreateTab("Executor Info", 0)
execTab:CreateLabel("Executor: " .. (identifyexecutor and identifyexecutor() or "Unknown"), 0, Color3.fromRGB(255,255,255), false)

-- User Info Tab (account age, premium status)
local userTab = Window:CreateTab("User Info", 0)
userTab:CreateLabel("Account Age: " .. player.AccountAge .. " days", 0, Color3.fromRGB(255,255,255), false)
local hasPremium = (player.MembershipType == Enum.MembershipType.Premium)
userTab:CreateLabel("Premium: " .. (hasPremium and "Yes" or "No"), 0, Color3.fromRGB(255,255,255), false)

-- Basic Needs Tab (WalkSpeed, JumpPower, Fly)
local basicTab = Window:CreateTab("Basic Needs", 0)
basicTab:CreateSlider({
    Name = "WalkSpeed",
    Range = {0, 300},
    Increment = 1,
    Suffix = "Speed",
    CurrentValue = humanoid.WalkSpeed,
    Flag = "WalkSpeed",
    Callback = function(value)
        humanoid.WalkSpeed = value
    end,
})
basicTab:CreateSlider({
    Name = "JumpPower",
    Range = {0, 200},
    Increment = 1,
    Suffix = "Power",
    CurrentValue = humanoid.JumpPower,
    Flag = "JumpPower",
    Callback = function(value)
        humanoid.JumpPower = value
    end,
})
basicTab:CreateButton({
    Name = "Fly Script",
    Callback = function()
        loadstring(game:HttpGet("https://raw.githubusercontent.com/XNEOFF/FlyGuiV3/main/FlyGuiV3.txt", true))()
    end,
})

-- Auto-run VW-Add loader after password is verified
loadstring(game:HttpGet("https://raw.githubusercontent.com/VapeVoidware/VW-Add/main/loader.lua", true))()
